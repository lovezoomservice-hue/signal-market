获取实时事件 JavaScript
// 初始化客户端 const client = new SignalMarket('YOUR_API_KEY'); // 获取事件列表 const events = await client.events.list({ stage: 'forming', limit: 10 }); events.forEach(event => { console.log(`${event.topic}: ${event.stage}`); });
订阅实时推送 JavaScript
// 订阅 WebSocket 事件流 const ws = client.subscribe('events'); ws.on('message', (event) => { console.log(`新事件: ${event.topic}`); // 自动处理 if (event.probability_pct > 0.8) { sendAlert(event); } });
Python 异步示例 Python
# 异步获取预测数据 import asyncio import signalmarket async def main(): client = signalmarket.Client('YOUR_API_KEY') # 并发请求多个 Lens results = await asyncio.gather( client.lenses.get_brief('lens_a_stock'), client.lenses.get_brief('lens_crypto_event'), ) for brief in results: print(f"Events: {len(brief.events)}") asyncio.run(main())
cURL 快速测试 cURL
# 获取所有活跃事件 curl -X GET "https://api.signalmarket.io/v1/events" \ -H "Authorization: Bearer YOUR_API_KEY" \ -H "Content-Type: application/json" # 筛选特定阶段的事件 curl -X GET "https://api.signalmarket.io/v1/events?stage=forming&limit=5" \ -H "Authorization: Bearer YOUR_API_KEY"
